function singleNumber(nums: number[]): number {
  let temp = nums.sort();
  if (temp[0] !== temp[1]) {
    return temp[0];
  }
  for (let i = 0; i < temp.length - 1; i++) {
    if (temp[i] === temp[i + 1]) {
      continue;
    }
    if (temp[i + 1] === temp[i + 2]) {
      continue;
    }
    return temp[i + 1];
  }
}

function singleNumber(nums: number[]): number {
  let result = 0;
  for (let i = 0; i < nums.length; i++) {
    result ^= nums[i];
  }
  return result;
}
